\newcommand{\getAvg}[3]{
	\pgfplotstableread[col sep=comma]{\Folder#1/#3#2#1_run1_Log.csv}\datafile
    	\pgfplotstablegetelem{0}{RenderAvg}\of\datafile
	\pgfmathsetmacro{\testValA}{\pgfplotsretval}
	\pgfplotstableread[col sep=comma]{\Folder#1/#3#2#1_run2_Log.csv}\datafile
  	\pgfplotstablegetelem{0}{RenderAvg}\of\datafile
	\pgfmathsetmacro{\testValB}{\pgfplotsretval}
% 	\pgfplotstableread[col sep=comma]{\Folder#1/#3#2#1_run3_Log.csv}\datafile
%     	\pgfplotstablegetelem{0}{RenderAvg}\of\datafile
% 	\pgfmathsetmacro{\testValC}{\pgfplotsretval}
	\pgfmathsetmacro{\temp}{(\testValA+\testValB)/2.0}
	
	\storelabel{#3#2#1}{\temp}
}
